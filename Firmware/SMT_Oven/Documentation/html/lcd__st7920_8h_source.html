244  void writeImage(const uint8_t *dataPtr, int x, int y, int width, int height) {
245  int offset = x&0x07;
246  int offsetPlusWidth = ((x+width-1)&0x07)+1;
247  int startMask = (uint8_t)(0xFF>>offset);
248  int endMask = (uint8_t)(0xFF00>>offsetPlusWidth);
249  // printf("x=[%d..%d], y=%d, w=%d, sm=0x%02X, em=0x%02X\n", x, x+width-1, y, width, startMask, endMask); fflush(stdout);
250 
251  for (int yy=y; yy<y+height; yy++) {
252  int xx = (yy*LCD_WIDTH)+x;
253  int ww = width;
254  int mask = startMask;
255  const uint8_t *dataByte = dataPtr;
256  int data = *dataByte++^invertMask;
257  if ((ww+offset)>8) {
258  // printf("+m=0x%02X\n", mask);
259  ww -= 8-(offset&0x07);
260  frameBuffer[xx/8] = (frameBuffer[xx/8]&~mask)|((data>>offset)&mask);
261  mask = 0xFF;
262  data = (data<<8)|(*dataByte++^invertMask);
263  xx += 8-(offset&0x07);
264  }
265  while (ww>8) {
266  // printf("=m=0x%02X\n", mask);
267  ww -= 8;
268  frameBuffer[xx/8] = (uint8_t)(data>>offset);
269  data = (data<<8)|(*dataByte++^invertMask);
270  xx += 8;
271  }
272  mask &= endMask;
273  // printf("-m=0x%02X\n", mask);
274  frameBuffer[xx/8] = (frameBuffer[xx/8]&~mask)|((data>>offset)&mask);
275  dataPtr += (width+7)/8;
276  }
277  // refreshImage(); // Debug only
278  }
