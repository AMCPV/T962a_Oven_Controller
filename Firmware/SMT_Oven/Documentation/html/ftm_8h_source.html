247  static ErrorCode setPeriod(float period) {
248  float inputClock = Info::getInputClockFrequency();
249  int prescaleFactor=1;
250  int prescalerValue=0;
251  while (prescalerValue<=7) {
252  float clock = inputClock/prescaleFactor;
253  uint32_t mod = round(period*clock);
254  if (mod < Info::minimumResolution) {
255  // Too short a period for 1% resolution
256  return setErrorCode(E_TOO_SMALL);
257  }
258  if (mod <= 65535) {
259  // Clear SC so immediate effect on prescale change
260  uint32_t sc = tmr->SC&~FTM_SC_PS_MASK;
261  tmr->SC = 0;
262  __DSB();
263  tmr->MOD = mod;
264  tmr->SC = sc|FTM_SC_PS(prescalerValue);
265  return E_NO_ERROR;
266  }
267  prescalerValue++;
268  prescaleFactor <<= 1;
269  }
270  // Too long a period
271  return setErrorCode(E_TOO_LARGE);
272  }
