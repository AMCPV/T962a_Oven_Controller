145  // Update input samples & error
146  lastInput = currentInput;
147  currentInput = inputFn();
148  currentError = setpoint - currentInput;
149 
150  integral += (ki * currentError);
151  if(integral > outMax) {

154  else if(integral < outMin) {
155  integral = outMin;
156  }
157  double dInput = (currentInput - lastInput);
158 
159  currentOutput = kp * currentError + integral - kd * dInput;
160  if(currentOutput > outMax) {
161  currentOutput = outMax;
162  }
163  else if(currentOutput < outMin) {

165  }
